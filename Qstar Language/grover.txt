|| grover || (circ oracle, qreg qr, qreg aux, int iters)
pre {iters=2}
	circ qr, aux ->
        init(qr,aux)
        assert{true}
        for i in range(iters) {
            invariant{true}
            grover_iter(oracle,qr,aux)
            assert {true}
        }
        return
pos {true}

| grover_iter | (circ oracle, qreg qr, qreg aux)
pre {true}
    circ qr, aux ->
        oracle(qr)
        diffusor(qr,aux)
        assert{true}
        return
pos {true}


| init | (qreg qr, qreg aux)
pre {true}
    circ qr, aux ->
        H(qr[:-1])
        X(aux)
        assert {true}
        H(aux)
        assert {true}
        return
pos {true}

| diffusor | (qreg qr, qreg aux)
pre {true}
    circ qr, aux ->
        with conjugated (H(qr)) {
            with conjugated (X(qr)) {
                with control qr[:-1] (Z(qr[-1]))
                assert {true}
            }
        }
        return
pos {true}
